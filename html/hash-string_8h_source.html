21 /* We assume to have `unsigned long int' value with at least 32 bits. */
22 #define HASHWORDBITS 32
23 
24 
25 /* Defines the so called `hashpjw' function by P.J. Weinberger
26  [see Aho/Sethi/Ullman, COMPILERS: Principles, Techniques and Tools,
27  1986, 1987 Bell Telephone Laboratories, Inc.] */
28 static unsigned long int
29 hash_string (const char *str_param)
30 {
31  unsigned long int hval, g;
32  const char *str = str_param;
33 
34  /* Compute the hash value for the given string. */
35  hval = 0;
36  while (*str != '\0')
37  {
38  hval <<= 4;
39  hval += (unsigned char) *str++;
40  g = hval & ((unsigned long int) 0xf << (HASHWORDBITS - 4));
41  if (g != 0)
42  {
43  hval ^= g >> (HASHWORDBITS - 8);
44  hval ^= g;
45  }
46  }
47  return hval;
48 }
