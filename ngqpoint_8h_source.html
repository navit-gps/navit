4static void get_direction(char *buffer, int angle, int mode) {
5 angle = angle % 360;
6 switch (mode) {
7 case 0:
8 sprintf(buffer, "%d", angle);
9 break;
10 case 1:
11 if (angle < 69 || angle > 291)
12 *buffer++ = 'N';
13 if (angle > 111 && angle < 249)

20 break;
21 case 2:
